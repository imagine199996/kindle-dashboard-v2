async function fetchAndRender() { try { const response = await fetch('/api/all-data'); const data = await response.json(); // 渲染新闻：加了 (data.news || [])，保证哪怕后端传空，前端也不会报错 document.getElementById('news-content').innerHTML = (data.news || []).map(n => `

${n.title}

`).join(''); // 渲染股票 document.getElementById('finviz-content').innerHTML = (data.finviz || []).map(s => `

${s.ticker}: ${s.change}

`).join(''); // 渲染天气 document.getElementById('weather-content').innerHTML = (data.weather || []).map(w => `

${w.name}: ${w.temp}

`).join(''); } catch (e) { console.error("Fetch failed", e); } }
